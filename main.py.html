#10 by ten rows of spots, dot= 20 in size, spaced around by 50
turtle.colormode(255)
t = Turtle()
t.hideturtle()

def row():
    for _ in range(10):
        color_rand = random.choice(color_list)
        t.pen()
        t.fillcolor(color_rand)
        t.pencolor(color_rand)
        t.dot(20)
        t.pu()
        t.forward(50)
        t.pd()

for _ in range(5):
    row()
    t.pu()
    t.backward(50)

    t.pd()
